md5

import hashlib
if __name__ == '__main__':
    str = "hello"
    hmd5 = hashlib.md5()
    # update之前必须encode
    hmd5.update(str.encode('utf-9'))
    print('加密前：' + str)
    print('加密后：' + hmd5.hexdigest())

工具

pip3 包管理工具

其他类库

xlrd : python读取excel

import xlrd

if __name__ == '__main__':
    data = xlrd.open_workbook("/tmp/test.xlsx")
    sheet = data.sheets()[0]
    nrows = sheet.nrows
    ncols = sheet.ncols
    for rowNum in range(0, nrows):
        row = sheet.row_values(rowNum)
        for colNum in range(0, ncols):
            cell = row[colNum]
            print(cell)
